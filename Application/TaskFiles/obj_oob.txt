echo This test is used to see if the engine handles objects out of bounds
echo usually, the culprit is a pointer issue
echo
echo
setRes 1000 1000
echo
echo Test 1: no Invokes
echo
echo spawning two inside bounds
spawn ./Resources/Renderobjects/standard.json|posX=100|posY=100
spawn ./Resources/Renderobjects/standard.json|posX=100|posY=500
echo spawning two outside bounds
spawn ./Resources/Renderobjects/standard.json|posX=2000|posY=100
spawn ./Resources/Renderobjects/standard.json|posX=2000|posY=500
wait 60
echo
echo done!
env-deload
wait 60
echo
echo Test 2: with Invokes
echo spawning two inside bounds
spawn ./Resources/Renderobjects/Planets/obj1.json|posX=100|posY=100|physics.vX=0|physics.vY=200
spawn ./Resources/Renderobjects/Planets/obj1.json|posX=100|posY=500|physics.vX=0|physics.vY=200
echo spawning two outside bounds
spawn ./Resources/Renderobjects/Planets/obj1.json|posX=2000|posY=100|physics.vX=0|physics.vY=0
spawn ./Resources/Renderobjects/Planets/obj1.json|posX=2000|posY=500|physics.vX=0|physics.vY=0
echo Waiting for Objects to move to new position...
wait 1000
echo
echo done!
env-deload
echo
echo
echo Test Passed
exit
